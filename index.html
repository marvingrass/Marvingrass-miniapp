from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup from telegram.ext import Updater, CommandHandler, MessageHandler, Filters, CallbackContext # Укажите ваш токен, полученный от BotFather BOT_TOKEN = "ВАШ_ТОКЕН_БОТА" # Укажите ID вашего Telegram-чата (можно сначала отправить сообщение боту и узнать ID через API) ADMIN_CHAT_ID = "ВАШ_ID_ТЕЛЕГРАМА" # Функция для обработки команды /start def start(update: Update, context: CallbackContext): update.message.reply_text( "Здравствуйте! Добро пожаловать в наш обменный пункт.\n" "Пожалуйста, отправьте запрос на обмен валюты в формате:\n" "USD 100 на EUR\n" "Или нажмите /help для получения инструкций." ) # Функция для обработки сообщения с заявкой def process_request(update: Update, context: CallbackContext): # Получение текста сообщения от пользователя user_message = update.message.text # Отправляем текст сообщения администратору context.bot.send_message( chat_id=ADMIN_CHAT_ID, text=f"Новая заявка на обмен от @{update.message.from_user.username or update.message.from_user.first_name}:\n{user_message}" ) # Отправляем подтверждающее сообщение пользователю update.message.reply_text("Ваша заявка отправлена. Мы свяжемся с вами в ближайшее время!") # Функция для обработки команды /help def help_command(update: Update, context: CallbackContext): update.message.reply_text( "Отправьте сообщение в формате:\n" "USD 100 на EUR\n" "Мы свяжемся с вами для дальнейших инструкций." ) def main(): # Создаем объект Updater и передаем ему токен updater = Updater(BOT_TOKEN, use_context=True) # Получаем ссылку на диспетчер, чтобы добавить обработчики dispatcher = updater.dispatcher # Обработчики команд dispatcher.add_handler(CommandHandler("start", start)) dispatcher.add_handler(CommandHandler("help", help_command)) # Обработчик текстовых сообщений (заявок) dispatcher.add_handler(MessageHandler(Filters.text & ~Filters.command, process_request)) # Запуск бота updater.start_polling() print("Бот запущен и работает...") updater.idle() if __name__ == "__main__": main()
